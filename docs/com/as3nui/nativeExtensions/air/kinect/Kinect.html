The Kinect class allows you to access the Kinect hardware in your AIR applications.

An instance of this class will give you access to the different video streams and skeleton information of a Kinect sensor.

Examples
BasicSkeletonExample.as

The following example checks if there is a Kinect device available, and renders the user's skeleton joints on the screen. 
 
package
{
    import com.as3nui.nativeExtensions.air.kinect.Kinect;
    import com.as3nui.nativeExtensions.air.kinect.KinectConfig;
    import com.as3nui.nativeExtensions.air.kinect.data.SkeletonJoint;
    import com.as3nui.nativeExtensions.air.kinect.data.User;
    
    import flash.display.Sprite;
    import flash.display.StageAlign;
    import flash.display.StageScaleMode;
    import flash.events.Event;
    import flash.geom.Point;
    import flash.geom.Vector3D;
    
    public class BasicSkeletonExample extends Sprite
    {
        public static const KinectMaxDepthInFlash:uint = 200;
        
        private var kinect:Kinect;
        private var skeletonContainer:Sprite;
        
        public function BasicSkeletonExample()
        {
            stage.align = StageAlign.TOP_LEFT;
            stage.scaleMode = StageScaleMode.NO_SCALE;
            if(Kinect.isSupported())
            {
                kinect = Kinect.getKinect();
                
                skeletonContainer = new Sprite();
                addChild(skeletonContainer);
                
                var config:KinectConfig = new KinectConfig();
                config.skeletonEnabled = true;
                kinect.start(config);
                
                addEventListener(Event.ENTER_FRAME, enterFrameHandler, false, 0, true);
            }
            stage.addEventListener(Event.RESIZE, resizeHandler, false, 0, true);
            resizeHandler();
        }
        
        protected function resizeHandler(event:Event = null):void
        {
            if(root != null)
            {
                root.transform.perspectiveProjection.projectionCenter = new Point(stage.stageWidth * .5, stage.stageHeight * .5);
            }
        }
        
        protected function enterFrameHandler(event:Event):void
        {
            skeletonContainer.removeChildren();
            for each(var user:User in kinect.users)
            {
                if(user.hasSkeleton)
                {
                    for each(var joint:SkeletonJoint in user.skeletonJoints)
                    {
                        if(joint.positionConfidence > .5)
                        {
                            skeletonContainer.addChild(createCircleForPosition(joint.positionRelative, 0x0000FF));
                        }
                    }
                }
                skeletonContainer.addChild(createCircleForPosition(user.positionRelative, 0xFF0000));
            }
        }
        
        private function createCircleForPosition(positionRelative:Vector3D, color:uint):Sprite
        {
            var circle:Sprite = new Sprite();
            circle.graphics.beginFill(color);
            circle.graphics.drawCircle(0, 0, 15);
            circle.graphics.endFill();
            circle.x = ((positionRelative.x + 1) * .5) * stage.stageWidth;
            circle.y = ((positionRelative.y - 1) / -2) * stage.stageHeight;
            circle.z = positionRelative.z * KinectMaxDepthInFlash;
            return circle;
        }
    }
}
BasicCameraExample.as

The following example, shows you how to display the RGB camera stream. 
 
package
{
    import com.as3nui.nativeExtensions.air.kinect.Kinect;
    import com.as3nui.nativeExtensions.air.kinect.KinectConfig;
    import com.as3nui.nativeExtensions.air.kinect.events.CameraImageEvent;
    
    import flash.display.Bitmap;
    import flash.display.Sprite;
    import flash.display.StageAlign;
    import flash.display.StageScaleMode;
    
    public class BasicCameraExample extends Sprite
    {
        
        private var rgbBitmap:Bitmap;
        private var kinect:Kinect;
        
        public function BasicCameraExample()
        {
            stage.align = StageAlign.TOP_LEFT;
            stage.scaleMode = StageScaleMode.NO_SCALE;
            if(Kinect.isSupported())
            {
                kinect = Kinect.getKinect();
                
                rgbBitmap = new Bitmap();
                addChild(rgbBitmap);
                
                kinect.addEventListener(CameraImageEvent.RGB_IMAGE_UPDATE, rgbImageUpdateHandler, false, 0, true);
                
                var config:KinectConfig = new KinectConfig();
                config.rgbEnabled = true;
                config.rgbWidth = 640;
                config.rgbHeight = 480;
                
                kinect.start(config);
            }
        }
        
        protected function rgbImageUpdateHandler(event:CameraImageEvent):void
        {
            rgbBitmap.bitmapData = event.imageData;
        }
    }
}
